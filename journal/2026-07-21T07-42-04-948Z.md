  'NEQ':   { op: OP.NEQ,   args: 0 },
  'LT':    { op: OP.LT,    args: 0 },
  'GT':    { op: OP.GT,    args: 0 },

  'CALL':  { op: OP.CALL,  args: 'lbl' },
  'RET':   { op: OP.RET,   args: 0 },
  'PRINT': { op: OP.PRINT, args: 0 },
};

export function assemble(source) {
  const lines = source.split('\n');
  const tokens = []; // flat list of { type, value }
  const labels = {};
  
  // First pass: tokenize and record label positions
  let pos = 0;
  for (const raw of lines) {
    const line = raw.split(';')[0].trim();
    if (!line) continue;
    
    if (line.endsWith(':')) {
      // label definition
      labels[line.slice(0, -1)] = pos;
      continue;
    }
    
    const parts = line.split(/\s+/);
    const mnemonic = parts[0].toUpperCase();
    const def = MNEMONICS[mnemonic];
    
    if (!def) throw new Error('Unknown mnemonic: ' + mnemonic);
    
    tokens.push({ mnemonic, def, operand: parts[1] });
    pos += def.args === 0 ? 1 : 2; // opcode + optional operand
  }
  
  // Second pass: emit bytecode
  const bytecode = [];
  for (const { mnemonic, def, operand } of tokens) {
    bytecode.push(def.op);
    if (def.args === 'num') {
      bytecode.push(Number(operand));
    } else if (def.args === 'lbl') {
      if (!(operand in labels)) throw new Error('Undefined label: ' + operand);
      bytecode.push(labels[operand]);
    } else if (def.args === 'sym') {
      bytecode.push(operand);
    }
    // args === 0: no operand
  }
  
  return bytecode;
}
